<script>
    // Real-time Clock
    setInterval(() => {
        document.getElementById('clock').innerText = new Date().toLocaleTimeString('vi-VN');
    }, 1000);

    // Chart.js Setup
    const ctx = document.getElementById('mainChart').getContext('2d');
    const myChart = new Chart(ctx, {
        type: 'bar',
        data: {
            labels: ['Đà Nẵng', 'Hà Nội', 'TP. HCM'],
            datasets: [{
                label: 'Khả năng tiết kiệm (Triệu VNĐ)',
                data: [0, 0, 0],
                backgroundColor: ['#10b981', '#6366f1', '#f59e0b'],
                borderRadius: 10,
                barThickness: 40
            }]
        },
options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: { legend: { display: false } },
            scales: {
                y: { grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#94a3b8' } },
                x: { grid: { display: false }, ticks: { color: '#94a3b8' } }
            }
        }
    });

    // Logic Phân tích & Insight
    function analyzeData() {
        const salary = Number(document.getElementById('salary').value);
        const otherCost = Number(document.getElementById('otherCost').value);
        const dnCost = 9000000; // Mức sống TB tại ĐN 2026 cho 1 IT-er

        const extra = (otherCost - dnCost);
        const savingsDN = (salary - dnCost);
        const savingsOther = (salary - otherCost);
        const savePercent = Math.min(((savingsDN / salary) * 100), 100).toFixed(0);

        // Hiển thị khung terminal
        document.getElementById('insight-terminal').style.display = 'block';
        document.getElementById('extra').innerText = `${(extra/1000000).toFixed(1)}Tr`;
        document.getElementById('dnCost').innerText = `${(dnCost/1000000).toFixed(1)}Tr`;

        // Hiệu ứng Typewriter cho Insight
        const insightText = document.getElementById('insight-text');
        const message = `Dựa trên dữ liệu: Mỗi tháng bạn sẽ dư thêm <b>${extra.toLocaleString()}đ</b> khi làm việc tại Đà Nẵng. <br><br>🔥 <b>Chiến lược:</b> Với số tiền này, bạn có thể trả góp một căn hộ Ocean-View hoặc nâng cấp bộ gear cực khủng. Chỉ số WLB 95% khẳng định bạn sẽ có thêm ít nhất 2 giờ mỗi ngày cho bản thân thay vì kẹt xe.`;
        
        insightText.innerHTML = ""; 
        let i = 0;
        function typeWriter() {
            if (i < message.length) {
                if(message.charAt(i) === '<') { 
                    let tagEnd = message.indexOf('>', i);
                    insightText.innerHTML += message.substring(i, tagEnd + 1);
                    i = tagEnd + 1;
                } else {
                    insightText.innerHTML += message.charAt(i);
                    i++;
                }
                setTimeout(typeWriter, 10);
            }
        }
        typeWriter();

        // Chạy thanh progress bar
        setTimeout(() => {
            document.getElementById('bar-save').style.width = savePercent + "%";
            document.getElementById('save-percent').innerText = savePercent + "%";
        }, 200);

        // Update Chart
        myChart.data.datasets[0].data = [savingsDN/1000000, savingsOther/1000000, (savingsOther*0.95)/1000000];
        myChart.update();
    }

    // AI Chat Logic
    function handleChat() {
        const input = document.getElementById('userInput');
        const chatFlow = document.getElementById('chatFlow');
        const val = input.value.trim();
if(!val) return;

        // User Message
        const userMsg = document.createElement('div');
        userMsg.className = 'bubble user';
        userMsg.innerText = val;
        chatFlow.appendChild(userMsg);
        input.value = '';

        // Bot Response
        setTimeout(() => {
            const botMsg = document.createElement('div');
            botMsg.className = 'bubble bot';
            
            let reply = "Tôi đang phân tích yêu cầu của bạn...";
            const lowVal = val.toLowerCase();
            if(lowVal.includes("lương")) reply = "Năm 2026, Senior IT tại Đà Nẵng có dải lương từ 45 - 75 triệu VNĐ. Các vị trí AI Engineer có thể chạm mốc 100 triệu.";
            else if(lowVal.includes("công ty")) reply = "Bạn nên nhắm vào các Hub lớn: FPT Software (Hòa Lạc Đà Nẵng), LG VS DCV (tòa nhà Indochina) hoặc các công ty Nhật tại bãi biển Mỹ Khê.";
            else if(lowVal.includes("đời sống") || lowVal.includes("ăn")) reply = "Đà Nẵng 2026 vẫn là thiên đường ẩm thực. Chi phí ăn uống chỉ bằng 70% so với Quận 1 TP.HCM.";
            else reply = "Đà Nẵng đang là 'Silicon Valley' của Việt Nam với chính sách miễn thuế cho nhân tài công nghệ cao đấy!";

            botMsg.innerHTML = reply;
            chatFlow.appendChild(botMsg);
            chatFlow.scrollTop = chatFlow.scrollHeight;
        }, 600);
    }
</script>
